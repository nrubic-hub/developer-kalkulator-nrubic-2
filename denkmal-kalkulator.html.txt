    <script>
        // JavaScript-Teil: Definiert die Berechnungslogik
        function format(value) {
            return value.toLocaleString('de-DE', { minimumFractionDigits: 0, maximumFractionDigits: 0 });
        }

        function berechneKalkulation() {
            const kaufpreis = parseFloat(document.getElementById('kaufpreis').value);
            const grundstuecksanteil_prozent = parseFloat(document.getElementById('grundstuecksanteil').value) / 100;
            const sanierungskosten_prozent = parseFloat(document.getElementById('sanierungskosten_prozent').value) / 100;
            const baujahr_vor_1925 = document.getElementById('baujahr_vor_1925').value === 'true';
            const kaltmiete_monatlich = parseFloat(document.getElementById('kaltmiete_monatlich').value);
            const grenzsteuersatz = parseFloat(document.getElementById('grenzsteuersatz').value) / 100;
            let kreditbetrag = parseFloat(document.getElementById('kreditbetrag').value);
            const sollzins_prozent = parseFloat(document.getElementById('sollzins').value) / 100;
            const tilgung_prozent = parseFloat(document.getElementById('tilgung').value) / 100;
            const nebenkosten_pa = parseFloat(document.getElementById('nebenkosten_pa').value);

            if (sanierungskosten_prozent + grundstuecksanteil_prozent >= 1) {
                alert("Grundstücks- und Sanierungskostenanteil dürfen zusammen 100% des Kaufpreises nicht überschreiten.");
                return;
            }

            const grundstuecksBasis = kaufpreis * grundstuecksanteil_prozent;
            const sanierungBasis = kaufpreis * sanierungskosten_prozent;
            const gebaeudeBasis = kaufpreis - grundstuecksBasis;
            const altbauBasis = gebaeudeBasis - sanierungBasis; 
            
            const altbauAfaRate = baujahr_vor_1925 ? 0.025 : 0.02; 
            const altbauAfaEuro = altbauBasis * altbauAfaRate;
            
            const kaltmiete_pa = kaltmiete_monatlich * 12;
            const anfaenglicheAnnunitaet = kreditbetrag * (sollzins_prozent + tilgung_prozent);

            let htmlTabelle = '';
            let kumulierteSteuerersparnis = 0;
            let kumulierterCashflow = 0;

            for (let jahr = 1; jahr <= 12; jahr++) {
                const zinsenEuro = kreditbetrag * sollzins_prozent;
                const tilgungEuro = anfaenglicheAnnunitaet - zinsenEuro;
                const restschuld = kreditbetrag - tilgungEuro;

                let sanierungAfaRate = 0;
                if (jahr <= 8) {
                    sanierungAfaRate = 0.09; 
                } else if (jahr <= 12) {
                    sanierungAfaRate = 0.07; 
                }
                const sanierungAfaEuro = sanierungBasis * sanierungAfaRate;

                const gesammtAfaEuro = altbauAfaEuro + sanierungAfaEuro;
                const werbungskosten = gesammtAfaEuro + zinsenEuro + nebenkosten_pa; 
                
                const einkommen_vermietung = kaltmiete_pa - werbungskosten;
                
                const steuerersparnis = (einkommen_vermietung < 0) ? Math.abs(einkommen_vermietung) * grenzsteuersatz : 0;
                
                const cashflowVorSteuern = kaltmiete_pa - anfaenglicheAnnunitaet - nebenkosten_pa;
                const cashflowNachSteuern = cashflowVorSteuern + steuerersparnis;

                kumulierteSteuerersparnis += steuerersparnis;
                kumulierterCashflow += cashflowNachSteuern;

                htmlTabelle += `
                    <tr>
                        <td>${jahr}</td>
                        <td>${format(restschuld)} €</td>
                        <td>${format(zinsenEuro)} €</td>
                        <td>${format(altbauAfaEuro)} €</td>
                        <td>${format(sanierungAfaEuro)} €</td>
                        <td class="${steuerersparnis > 0 ? 'positive' : ''}">${format(steuerersparnis)} €</td>
                        <td class="${cashflowNachSteuern > 0 ? 'positive' : 'negative'}">${format(cashflowNachSteuern)} €</td>
                    </tr>
                `;

                kreditbetrag = restschuld;
            }

            document.getElementById('ergebnis-tabelle').innerHTML = htmlTabelle;
            
            document.getElementById('ersparnis-gesamt').textContent = format(kumulierteSteuerersparnis) + ' €';
            document.getElementById('ersparnis-gesamt').className = kumulierteSteuerersparnis > 0 ? 'positive' : 'negative';

            document.getElementById('cashflow-kumuliert').textContent = format(kumulierterCashflow) + ' €';
            document.getElementById('cashflow-kumuliert').className = kumulierterCashflow > 0 ? 'positive' : 'negative';
            
            document.getElementById('ergebnis-section').style.display = 'block';
        }

        window.onload = berechneKalkulation;
    </script>
